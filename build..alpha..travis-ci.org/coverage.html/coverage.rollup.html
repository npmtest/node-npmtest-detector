function detect(name, expression, ua) {
  const expr = isFunction(expression) ? expression.call(null, ua) : expression;
  if (!expr) { return null; }
  const info = {
    name: name,
    version: NA_VERSION,
    codename: "",
  };
  if (expr === true) {
    return info;
  }else if(isString(expr)) {
    if(ua.indexOf(expr) !== -1){
      return info;
    }
  } else if (isObject(expr)) {
